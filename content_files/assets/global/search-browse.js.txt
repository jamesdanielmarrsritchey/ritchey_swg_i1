// Helper: fetch JSON with explicit error handling
async function fetch_json_or_throw_var(url_var) {
  var response_var = await fetch(url_var, { cache: "no-store" });

  if (!response_var.ok) {
    var error_var = new Error("HTTP " + response_var.status + " for " + url_var);
    error_var.status = response_var.status;
    error_var.url = url_var;
    throw error_var;
  }

  return await response_var.json();
}

async function get_database_items_v1() {

    return;

  try {
    // Determine which database to access
    var database_path_var;
    if (item_order_gvar == "reverse chronological") {
      database_path_var = address_gvar + "/databases/reverse_chronological/";
    } else {
      database_path_var = address_gvar + "/databases/chronological/";
    }

    // Local results array
    var results_arr = [];

    // Starting index
    var n_var = parseInt(item_range_garr[0], 10);

    // Total number of items (learned from first valid entry)
    var last_item_var = null;

    while (results_arr.length < items_per_page_gvar) {
      var json_url_var = database_path_var + n_var + ".json";
      n_var = n_var + 1;

      var entry_var;

      try {
        entry_var = await fetch_json_or_throw_var(json_url_var);
      } catch (error_var) {
        // Stop cleanly if file does not exist
        if (error_var && error_var.status === 404) {
          break;
        }

        // Any other error stops execution to prevent infinite loops
        console.log("Stopped due to error:", error_var);
        break;
      }

      // Learn total item count from first successful entry
      if (last_item_var === null) {
        var item_number_var = entry_var.item_number;

        if (typeof item_number_var === "string" && item_number_var.indexOf(" of ") !== -1) {
          var total_str_var = item_number_var.split(" of ")[1];
          last_item_var = Number(total_str_var);
        } else {
          console.log("Stopped: item_number format unexpected:", item_number_var);
          break;
        }
      }

      // Stop if we've processed all known items
      if (last_item_var !== null && n_var > (last_item_var + 1)) {
        break;
      }

      // Apply filtering criteria
      var keep_var = true;
	
	// Filter by type
	if (item_types_garr.includes(entry_var.item_type) !== true){
		keep_var = false;
		//console.log(item_types_garr);
		//console.log('Type Filter:' + entry_var.item_type);
	}
	// Filter by status
	if (item_statuses_garr.includes(entry_var.item_status) !== true){
		keep_var = false;
		//console.log(item_statuses_garr);
		//console.log('Status Filter:' + entry_var.item_status);
	}
	// Filter by keywords
	function check_if_array_empty_v1(array_arr) {
		if (array_arr.length === 0) {
			return true;
		}
		if (array_arr.length === 1) {
			if (array_arr[0] === null || array_arr[0] === undefined) {
				return true;
			}
			if (typeof array_arr[0] === 'string' && array_arr[0].trim() === '') {
				return true;
			}
		}
		return false;
	}
	
	if (check_if_array_empty_v1(item_keywords_garr) === false){
		function non_alnum_to_commas_v1(string_var) {
		return String(string_var)
			.toLowerCase()
			.replace(/[^a-z0-9]+/g, ',')
			.replace(/^,+|,+$/g, '');
		}
		var matches_found_var = 0;
		var entry_keywords_var = non_alnum_to_commas_v1(entry_var.item_name);
		var entry_keywords_arr = entry_keywords_var.split(",");
		for (var n2_var = 0; n2_var < item_keywords_garr.length; n2_var++) {
    			if (entry_keywords_arr.includes(item_keywords_garr[n2_var])) {
      	  			matches_found_var++;
    			}
		}
		//console.log(matches_found_var);

		// When this checks keywords it's calculating how many of the requested keywords are present in the source, but doesn't count how many other words are in the source that weren't requested.
		var item_keywords_match_percent_var = (item_keywords_match_gvar / 100).toFixed(2);
		var match_found_percent_var = (matches_found_var / item_keywords_garr.length).toFixed(2);
		console.log(match_found_percent_var);
		if (match_found_percent_var < item_keywords_match_percent_var){
			keep_var = false;
			//console.log(item_keywords_garr);
			//console.log('Percent Filter:' + match_found_percent_var);
		} else {
			//console.log(item_keywords_garr);
			//console.log('Percent Filter:' + match_found_percent_var);
			// This adds keyword match information to the entry_var object, since it doesn't come with the JSON file.
			entry_var.item_keyword_match_percent = match_found_percent_var * 100;
			entry_var.item_keyword_match_percent = entry_var.item_keyword_match_percent + '%';
		}
	} else {
		entry_var.item_keyword_match_percent = '100%'; // This is for when there's no keywords being searched.
	}

      if (keep_var === true) {
        results_arr.push(entry_var);
      }

      // Hard stop once last known item is processed
      if (last_item_var !== null && (n_var - 1) >= last_item_var) {
        break;
      }
    }

    // Update global array once loading is complete
    database_items_garr = results_arr;
	
	window.database_items_count_wvar = database_items_garr.length

	reset_results_gvar = true;
    console.log("Loaded entries:", database_items_garr.length);
    console.log(database_items_garr);
  } finally {
    database_loading_gvar = false;
  }
}

// Show, hide search controls, depending on whether in search, or browse mode. As well as show/hide keyword matching information within results.
function toggle_element_visibility_v1(element_id) {
    var el = document.getElementById(element_id);
    if (!el) return;

    if (el.style.display === "none") {
        el.style.display = "block";
    } else {
        el.style.display = "none";
    }
}
function set_element_display_v1(element_id, display_value) {
    var el = document.getElementById(element_id);
    if (!el) return;

    el.style.display = display_value;
}
function toggle_elements_by_class_v1(class_name) {
    var elements = document.getElementsByClassName(class_name);

    for (var i = 0; i < elements.length; i++) {
        if (elements[i].style.display === "none") {
            elements[i].style.display = "";
        } else {
            elements[i].style.display = "none";
        }
    }
}
function set_display_for_class_v1(class_name, display_value) {
    var elements = document.getElementsByClassName(class_name);

    for (var i = 0; i < elements.length; i++) {
        elements[i].style.display = display_value;
    }
}
function set_class_display_rule_v1(class_name, display_value) {
    var style_id = "dynamic_class_display_rules";
    var style_element = document.getElementById(style_id);

    if (!style_element) {
        style_element = document.createElement("style");
        style_element.id = style_id;
        document.head.appendChild(style_element);
    }

    var sheet = style_element.sheet;
    var selector = "." + class_name;
    var rule_text = selector + " { display: " + display_value + "; }";

    // Remove existing rule for this class if it exists
    for (var i = sheet.cssRules.length - 1; i >= 0; i--) {
        if (sheet.cssRules[i].selectorText === selector) {
            sheet.deleteRule(i);
        }
    }

    sheet.insertRule(rule_text, sheet.cssRules.length);
}

// Show, or hide the next/previous buttons, as well as update them.
function update_item_range_parameter_in_URL_string_v1(range_arr) {
	// Make sure we were given an array with at least 2 values
	if (!Array.isArray(range_arr) || range_arr.length < 2) {
		return false;
	}

	// Convert first two values to numbers
	var start_num = Number(range_arr[0]);
	var end_num = Number(range_arr[1]);

	// Validate numbers
	if (isNaN(start_num) || isNaN(end_num)) {
		return false;
	}

	// Create URL object from current location
	var url_obj = new URL(window.location.href);

	// Set or update item_range parameter
	url_obj.searchParams.set('item_range', start_num + ',' + end_num);

	// Return updated URL as a string (do NOT modify browser URL)
	return url_obj.toString();
}
